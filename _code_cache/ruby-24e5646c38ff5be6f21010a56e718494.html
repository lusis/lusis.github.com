class HTTPPool
  include Celluloid::Actor
  WORKERS = 10

  attr_reader :workers

  def initialize(name)
    @name = name
    @workers = []
    WORKERS.times do |id|
      @workers[id] = HTTPWorker.supervise_as "http_worker_#{id}".to_sym, "http_worker_#{id}"
    end
  end
  def do_work
    @workers.sample.actor.do_work "msg"
  end
end
